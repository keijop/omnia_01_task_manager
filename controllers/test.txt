CREATE TABLE tasks (
    taskid int NOT NULL AUTO_INCREMENT,
    userid int NOT NULL,
    description varchar(256) NOT NULL,
    completed bool NOT NULL,
    deadline date,
    PRIMARY KEY (taskid),
    FOREIGN KEY (userid) REFERENCES users(userid)
);
